# 짝수의 합
- 정수 n이 주어질 때, n이하의 짝수를 모두 더한 값을 return 하도록 solution 함수를 작성해주세요.



# 제한사항
- 0 < n ≤ 1000

# 입출력 예
| n | result |
| -- | ----- |
| 10 | 30|
| 4 | 6 |

# 입출력 예 설명
- n이 10이므로 2 + 4 + 6 + 8 + 10 = 30을 return 합니다.
- n이 4이므로 2 + 4 = 6을 return 합니다.

```javascript
unction solution(n) {
    let arr = [];
    let even = [];
    let sum;

    n = 10;

    // 1) n을 배열로 만들어준다.
    // 2) 배열에서 짝수를 찾는다.

    for (i=0;i <= n; i++) {
        arr.push(i);

        if (arr[i]%2 === 0) {
            
            even.push(i);
            console.log(even);

            sum = even.reduce((a,b) => (a+b));

            console.log(sum);
        }
    }
    console.log(sum);
    return sum;
}
```
