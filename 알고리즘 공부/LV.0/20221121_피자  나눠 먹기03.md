## 방법1
```javascript
function solution(slice, n) {
    let answer = 0;

    // 1) n은 인원 수, slice는 피자 조각 수
    // 2) slice 범위는 2~10 사이
    // 3) slice조각의 배수보다 n의 인원수가 많아 나머지가 생기면
    // 4) 올림을 한다

    answer = Math.ceil(n/slice); 

    return answer;
}
```
