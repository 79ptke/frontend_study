function solution(n) {
    let answer = 0;

    // 1) n은 인원 수, 7은 피자 조각 수
    // 2) 7 단위로 피자가 한 판씩 늘어남
    // 3) 7 이하는 1판, 8~14 사이는 2판, 15~21 사이는 3판  
    
    n = 15;
    let pizza = 7;

    // console.log(15 / 7); // 2.14
    // console.log(Math.floor(15 / 7)); // 2
    // console.log(15 % 7); // 나머지가 1
    let num = Math.floor(n/pizza);

    if(n%pizza > 0) {
        //num = Math.floor(n/pizza);

        answer = num + 1;

        console.log(answer);
    } else if(n%pizza === 0) {
        answer = num;
        console.log(answer);
    }

    return answer;
}
